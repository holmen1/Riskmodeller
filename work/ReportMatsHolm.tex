n_val<-1000

res_mat2<-matrix(0,runs,3)

	z_val<-rnorm(1)
	z_vec<-rnorm(n_val)
	res_mat2[i,1]<-sum(exp(z_vec))
	res_mat2[i,3]<-sum(exp(0.1*z_val+sqrt(1-0.1^2)*z_vec))
}

hist(res_mat2[,1],75,xlab="",ylab="",main="rho=0",xlim=c(min(res_mat2),max(res_mat2)),
col="black")
